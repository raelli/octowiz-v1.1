```python
def rewrite_namespace(memories: List[Dict[str, Any]], namespace: str) -> List[Dict[str, Any]]:
    result = []
    for m in memories:
        m = dict(m)
        m["key"] = (
            m["key"]
            .replace("team:allspark:", f"team:{namespace}:")
            .replace("project:allspark:", f"project:{namespace}:")
        )
        result.append(m)
    return result
```

```python
class TestLoadMemories(unittest.TestCase):
    def _write_temp(self, content, suffix):
        f = tempfile.NamedTemporaryFile(mode="w", suffix=suffix, delete=False)
        f.write(content)
        f.close()
        return f.name

    def test_load_json_list(self):
        data = [{"key": "k1", "value": "v1"}, {"key": "k2", "value": "v2"}]
        path = self._write_temp(json.dumps(data), ".json")
        try:
            self.assertEqual(load_memories(path), data)
        finally:
            os.unlink(path)

    def test_load_json_object_with_memories_key(self):
        data = {"memories": [{"key": "k1", "value": "v1"}]}
        path = self._write_temp(json.dumps(data), ".json")
        try:
            self.assertEqual(load_memories(path), data["memories"])
        finally:
            os.unlink(path)

    def test_load_jsonl(self):
        lines = '{"key": "k1", "value": "v1"}\n{"key": "k2", "value": "v2"}\n'
        path = self._write_temp(lines, ".jsonl")
        try:
            result = load_memories(path)
            self.assertEqual(len(result), 2)
            self.assertEqual(result[1]["key"], "k2")
        finally:
            os.unlink(path)
```

## Phase routing

3. `/mattpocock-skills:to-prd` — synthesise the conversation into a formal PRD on the issue tracker
4. `/superpowers:writing-plans` — may already have been invoked at the end of brainstorming; run standalone if not
5. `/mattpocock-skills:to-issues` — break the plan into independently-grabbable vertical slice issues

4. `/mattpocock-skills:to-issues` — vertical slice issues on the tracker

3. `/superpowers:executing-plans` — execute the written plan with review checkpoints

5. `/superpowers:verification-before-completion` — evidence-before-assertions gate; must pass before claiming done or opening a PR

## Doctrine reference

Key principles from the fetched memories (summarised for when memory fetch fails):

- **Context smart zone:** keep context windows small; split large work into focused tasks
- **HITL vs AFK:** alignment and product decisions stay human-in-the-loop; only well-scoped, testable, dependency-resolved tasks go AFK
- **Tracer bullets:** prefer thin vertical slices over horizontal layers; first slice should cross schema, service, and minimal UI
- **TDD:** write the failing test before implementation, always
- **Fresh context review:** never review in the same context window used for implementation
