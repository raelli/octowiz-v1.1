# Octowiz A2A v1 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add the Octowiz A2A skill to AELLI (`/a2a/octowiz`), ship the self-improve guard, and close Gap 2 in octowiz — producing a stable v1 of the Octowiz engineering workflow.

**Architecture:** New `src/skills/octowiz/` skill registered at `/a2a/octowiz`. Composes COMPACT/STANDARD/FULL context bundles from the MemPalace playbook (via HTTP memory client), git snapshot (shell), and LiteLLM doctrine (HTTP fallback when playbook is empty). Engineering Qdrant hits added on FULL tier. Self-improve guard is a standalone Node.js script in `bin/`. Gap 2 is a PR merge in the octowiz repo.

**Tech Stack:** Node.js 18+ (native fetch), Jest 29, Express (existing), `child_process.execSync` for git

```js
'use strict';

const BUNDLE_TIERS = Object.freeze({ COMPACT: 'COMPACT', STANDARD: 'STANDARD', FULL: 'FULL' });

const ROUTER_TO_BUNDLE = {
  SIMPLE: 'COMPACT',
  MEDIUM: 'COMPACT',
  COMPLEX: 'COMPACT',
  REASONING: 'STANDARD',
  NORMAL: 'FULL',
  HIGH: 'FULL',
  XHIGH: 'FULL',
  MAX: 'FULL',
};

function resolveBundleTier({ bundleTier, routerTier } = {}) {
  if (bundleTier && BUNDLE_TIERS[bundleTier]) return bundleTier;
  if (routerTier && ROUTER_TO_BUNDLE[routerTier]) return ROUTER_TO_BUNDLE[routerTier];
  return 'STANDARD';
}

module.exports = { resolveBundleTier, BUNDLE_TIERS };
```

```js
// src/skills/octowiz/bundle-builder.js
'use strict';

const path = require('path');
const { spawnSync } = require('child_process');
const { resolveBundleTier } = require('./model-tier');

const DOCTRINE_KEYS = {
  plan: 'agent%3Aplanner%3Amemory%3Aai-coding-workflow',
  review: 'agent%3Areviewer%3Amemory%3Aai-coding-workflow',
};

async function fetchFallbackDoctrine(operation) {
  const base = process.env.LITELLM_BASE_URL;
  const key = process.env.LITELLM_API_KEY || process.env.LITELLM_ADMIN_API_KEY || '';
  if (!base) return '';
  const docKey = DOCTRINE_KEYS[operation] || DOCTRINE_KEYS.plan;
  try {
    const res = await fetch(`${base}/v1/memory/${docKey}`, {
      headers: { Authorization: `Bearer ${key}` },
    });
    if (!res.ok) return '';
    const data = await res.json();
    return typeof data === 'string' ? data : (data.text || data.value || JSON.stringify(data));
  } catch {
    return '';
  }
}

function getGitSnapshot(repoRoot, maxFiles) {
  const root = path.resolve(repoRoot || process.cwd());
  try {
    const branchResult = spawnSync('git', ['-C', root, 'rev-parse', '--abbrev-ref', 'HEAD'], {
      encoding: 'utf8', stdio: ['pipe', 'pipe', 'pipe'],
    });
    const branch = branchResult.stdout.trim();
    const logResult = spawnSync('git', ['-C', root, 'log', '--name-only', '--pretty=format:', '-20'], {
      encoding: 'utf8', stdio: ['pipe', 'pipe', 'pipe'],
    });
    const recent_files = [...new Set(logResult.stdout.split('\n').map(l => l.trim()).filter(Boolean))].slice(0, maxFiles);
    return { branch, recent_files };
  } catch {
    return { branch: 'unknown', recent_files: [] };
  }
}

async function buildBundle({ bundleTier, routerTier, operation, scope, repoRoot, task, pgStore, memStore, engineering }) {
  const tier = resolveBundleTier({ bundleTier, routerTier });

  let playbook = '';
  if (pgStore) {
    try {
      const val = await pgStore.get(`agent:aelli:playbook:${scope}`);
      playbook = val ? String(val) : '';
    } catch { /* graceful degradation */ }
  }

  if (!playbook) {
    playbook = await fetchFallbackDoctrine(operation);
  }

  if (tier === 'COMPACT') {
    const snap = getGitSnapshot(repoRoot, 5);
    return {
      tier,
      scope,
      playbook: playbook.slice(0, 600),
      session: { branch: snap.branch, recent_files: snap.recent_files, active_task: null },
    };
  }

  if (tier === 'STANDARD') {
    const snap = getGitSnapshot(repoRoot, 10);
    return {
      tier,
      scope,
      playbook,
      session: { branch: snap.branch, recent_files: snap.recent_files, active_task: null, recent_events: [] },
    };
  }

  // FULL
  const snap = getGitSnapshot(repoRoot, 10);
  let knowledge = [];
  if (engineering && task) {
    try {
      const result = await engineering.handle({ type: 'engineering:query', query: task });
      knowledge = Array.isArray(result && result.hits) ? result.hits.slice(0, 5) : [];
    } catch { /* graceful degradation */ }
  }

  const bundle = {
    tier,
    scope,
    playbook,
    session: { branch: snap.branch, recent_files: snap.recent_files, active_task: null, recent_events: [] },
    knowledge,
  };

  if (operation === 'review' && memStore) {
    try {
      const rows = await memStore.getExperiencesSinceCursor(scope, 0, 100);
      const recent = rows.slice(-10);
      bundle.session.experiences_summary = recent
        .map(r => JSON.stringify(r.payload))
        .join('\n---\n')
        .slice(0, 2000);
    } catch {
      bundle.session.experiences_summary = '';
    }
  }

  return bundle;
}

module.exports = { buildBundle };
```

```js
// src/skills/octowiz/operations.js
'use strict';

const { buildBundle } = require('./bundle-builder');

function createOperations({ pgStore, memStore, engineering }) {
  async function plan({ scope = 'unknown', routerTier, bundleTier, task = '', repoRoot } = {}) {
    return buildBundle({ operation: 'plan', scope, routerTier, bundleTier, task, repoRoot, pgStore, memStore, engineering });
  }

  async function review({ scope = 'unknown', routerTier, bundleTier, task = '', repoRoot } = {}) {
    return buildBundle({ operation: 'review', scope, routerTier, bundleTier, task, repoRoot, pgStore, memStore, engineering });
  }

  return { plan, review };
}

module.exports = { createOperations };
```

```js
#!/usr/bin/env node
// bin/improve-guard.js
// Called by the self-improve CCR before git commit.
// Exits 1 (abort) if the diff exceeds the structural threshold.
// Exits 0 (allow) if the diff is small and focused.
'use strict';

const { execSync } = require('child_process');
const { appendFileSync, mkdirSync } = require('fs');
const path = require('path');

const THRESHOLD_LINES = 50;
const THRESHOLD_FILES = 2;
const QUEUE_FILE = path.join(process.cwd(), '.octowiz', 'improve-review-queue.jsonl');

function getDiffStats() {
  try {
    const output = execSync('git diff --cached --stat', { encoding: 'utf8', stdio: ['pipe', 'pipe', 'pipe'] });
    const lines = output.trim().split('\n');
    const summary = lines[lines.length - 1] || '';
    const filesMatch = summary.match(/(\d+) file/);
    const insertMatch = summary.match(/(\d+) insertion/);
    const deleteMatch = summary.match(/(\d+) deletion/);
    return {
      changedFiles: filesMatch ? parseInt(filesMatch[1], 10) : 0,
      changedLines: (insertMatch ? parseInt(insertMatch[1], 10) : 0) +
                    (deleteMatch ? parseInt(deleteMatch[1], 10) : 0),
    };
  } catch {
    return { changedFiles: 0, changedLines: 0 };
  }
}

function flagForReview(stats) {
  try {
    mkdirSync(path.dirname(QUEUE_FILE), { recursive: true });
    appendFileSync(QUEUE_FILE, JSON.stringify({ timestamp: new Date().toISOString(), ...stats }) + '\n');
  } catch (err) {
    process.stderr.write(`[improve-guard] could not write review queue: ${err.message}\n`);
  }
}

const stats = getDiffStats();
const exceeds = stats.changedLines > THRESHOLD_LINES || stats.changedFiles > THRESHOLD_FILES;

if (exceeds) {
  process.stderr.write(
    `[improve-guard] flagged: ${stats.changedFiles} files, ${stats.changedLines} lines ` +
    `(threshold: >${THRESHOLD_FILES} files OR >${THRESHOLD_LINES} lines)\n`
  );
  flagForReview(stats);
  process.exit(1);
}

process.stdout.write(`[improve-guard] ok: ${stats.changedFiles} files, ${stats.changedLines} lines\n`);
process.exit(0);
```
